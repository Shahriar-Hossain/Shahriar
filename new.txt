import java.util.*;
public class Sum {

public static void main(String[] args) {

Scanner sc= new Scanner(System.in);
System.out.println("\n Enter The Number:");
int n=sc.nextInt();

int sum =0;
for(int i=1; i<=n; i++) {
sum+=i;
}
System.out.println("\n All Sum Number is: "+sum);
if(sum%2==0) {
System.out.println("\n The Sum Number Is Even: "+sum);
}
else {
System.out.println("\n The Sum Number is Odd: "+sum);
}

}

}
